## 依赖配置 ##

```xml
<repositories>
  <repository>
    <id>lychie-maven-repo</id>
    <url>https://raw.github.com/lychie/maven-repo/master/releases</url>
  </repository>
</repositories>

<dependencies>
  <dependency>
    <groupId>org.lychie</groupId>
    <artifactId>jexcel</artifactId>
    <version>1.0.1</version>
  </dependency>
</dependencies>
```
